# ========== FORM SUBMISSION ==========
on form submit:
    youtube_url = get_input_url()

    if is_valid_youtube_url(youtube_url):
        send POST to /api/start-fetch with youtube_url


# ========== BACKEND: START FETCHING ==========
/api/start-fetch:
    cache_key = hash(youtube_url)

    # If cache exists:
    if cache_status[cache_key] exists:
        if cache_status[cache_key].status == "done":
            return cache_key  # Data is fully fetched and ready for use.
        elif cache_status[cache_key].status == "started":
            return cache_key  # Data is still being fetched, wait for completion.
        elif cache_status[cache_key].status == "expired":
            # Cache is expired, need to re-fetch the data
            reset cache_status[cache_key]  # Clear or reset relevant cache fields
            start background thread to fetch new data
            return cache_key  # Data will be fetched again

    else:
        # Initialize cache status for this task.
        initialize cache_status[cache_key] = {
            status: "started",
            progress: 0,
            video_ids: [],
            fetched_video_info_ids: [],
            completed: false,
            last_updated: now()
        }

        # Start background thread to fetch video IDs from YouTube.
        start background thread:
            result = yt_dlp.extract_flat(youtube_url)

            for every video_id in result.entries:
                append to cache_status[cache_key].video_ids
                update progress:
                    cache_status[cache_key].progress = (num_fetched / total)

                # If enough video links are fetched to fill 1 page of requested batch size, allow video info fetch.
                if enough to fill 1 page:
                    allow video_info fetch

            # Mark the task as completed once all data is fetched.
            when finished:
                cache_status[cache_key].status = "done"
                cache_status[cache_key].progress = 100
                cache_status[cache_key].completed = true
                cache_status[cache_key].last_updated = now()


# ========== VIDEO INFO FETCHING ==========
/api/video-info:
    input: cache_key, page_number, batch_size

    # Check if cache for the task exists and hasn't expired.
    if cache_status[cache_key] not found or expired:
        return error: "Cache expired" or loading: true

    # Calculate range of video IDs to fetch based on requested page and batch size.
    start = (page_number - 1) * batch_size
    end = start + batch_size
    video_ids_batch = cache_status[cache_key].video_ids[start:end]

    # Fetch video metadata for each video in the batch.
    for video_id in video_ids_batch:
        if video_id not in video_info_cache:
            metadata = yt_dlp.get_info(video_id)
            video_info_cache[video_id] = {
                title, thumbnail, embed_url, timestamp: now()
            }

            append video_id to cache_status[cache_key].fetched_video_info_ids

    return metadata list for that batch


# ========== LOAD MORE ==========
on "Load More" button click:
    # Check if the fetch task is still in progress.
    if cache_status[cache_key].status == "started":
        if next page’s worth of video_ids exist:
            call /api/video-info with next page
        else:
            deny fetch: not enough data yet

    # If the task is completed, allow loading of more pages.
    elif cache_status[cache_key].status == "done":
        if more pages available:
            call /api/video-info with next page
        else:
            show "no more videos" message


# ========== FRONTEND RENDERING ==========
on receiving video metadata batch:
    # Render each video element with metadata.
    for each video in batch:
        render element (title, thumbnail, embed)
